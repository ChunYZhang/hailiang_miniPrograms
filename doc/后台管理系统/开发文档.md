### 2.3 娃娃表（doll）

```sql
CREATE TABLE doll (
  id VARCHAR(36) PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  price DECIMAL(10,2) DEFAULT 0,
  material VARCHAR(50),
  size VARCHAR(50),
  stock INT DEFAULT 0,
  status ENUM('active','inactive') DEFAULT 'active',
  series VARCHAR(50),
  patent_no VARCHAR(100),
  description TEXT,
  images JSON,
  views INT DEFAULT 0,
  inquiries INT DEFAULT 0,
  low_stock_threshold INT DEFAULT 10,
  min_quantity INT DEFAULT 1 COMMENT '起购数量（箱）',
  small_box_capacity INT DEFAULT 500,
  medium_box_capacity INT DEFAULT 1000,
  large_box_capacity INT DEFAULT 2000,
  default_accessory VARCHAR(100),
  selected_accessories JSON COMMENT '已选配饰 [{id,name,price}]',
  is_hot TINYINT DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

### 2.4 购物车表（mini_cart）

```sql
CREATE TABLE mini_cart (
  id VARCHAR(36) PRIMARY KEY,

  quantity INT DEFAULT 1,
  pindan_group_id VARCHAR(36),
  pindan_group_name VARCHAR(100),
  box_size VARCHAR(20) DEFAULT 'small',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
```

```python
    cursor.execute("SELECT id FROM user WHERE phone=%s", (phone,))
    user = cursor.fetchone()
    if not user:
        # 自动注册
```

## 五、中间件与装饰器

### 5.1 后台管理员装饰器

```python
def admin_login_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('admin_token')
        if not token:
            return jsonify({"code": 401, "msg": "请先登录"})
        # 验证token，获取admin信息
        return f(*args, **kwargs)
    return decorated
```

### 5.2 小程序用户装饰器

```python
def mini_login_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('X-Token')
        if not token:
            return jsonify({"code": 401, "msg": "请先登录"})
        user_id = get_mini_user_id_from_token(token)
        if not user_id:
            return jsonify({"code": 401, "msg": "登录已过期"})
        return f(*args, **kwargs)
    return decorated
```
